```sql
--例如,批量添加100个字段
declare @temp int
set @temp = 0
declare @t nvarchar(255)

while @temp <=99
begin
select @t = 'str'+convert(nvarchar(255),@temp)
sp_msforeachtable ‘ALTER TABLE ? ADD @t int default 0 not null’ --添加字段
set @temp = @temp + 1
end

sp_msforeachtable ‘ALTER TABLE ? drop column columnName’–删除字段
```
